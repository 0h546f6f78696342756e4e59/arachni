def initialize( url, raw = {} )
    super( url, raw )

    @action = @url
    @method = 'cookie'

    @auditable = { @raw['name'] => @raw['value'] }
    @simple = @auditable.dup
    @auditable.reject! {
        |cookie|
        Options.instance.exclude_cookies.include?( cookie )
    }

    @orig      = @auditable.deep_clone
    @orig.freeze
end
