def audit_id( injection_str )
    vars = auditable.keys.sort.to_s
    url = URI( @auditor.page.url ).merge( URI( @action ).path ).to_s

    return "#{@auditor.class.info[:name]}:" +
      "#{url}:" + "#{self.type}:" +
      "#{vars}=#{injection_str.to_s}"
end
