def self.fill( hash )
    
    hash.keys.each{
        |key|
        
        next if hash[key] && !hash[key].empty?
        
        if val = self.match?( key )
            hash[key] = val
        end
        
        # moronic default value...
        # will figure  out ssomething better in the future...
        hash[key] = '1' if( !hash[key] || hash[key].empty? )
    }
    
    return hash
end
