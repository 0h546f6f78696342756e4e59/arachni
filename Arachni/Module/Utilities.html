def Utilities.get_path( url )
  
    splits = []
    tmp = ''
    
    url.each_char {
        |c|
        if( c != '/' )
            tmp += c
        else
            splits << tmp
            tmp = ''
        end
    }
    
    if( !tmp =~ /\./ )
      splits << tmp
    end
    
    return splits.join( "/" ) + '/'
end
