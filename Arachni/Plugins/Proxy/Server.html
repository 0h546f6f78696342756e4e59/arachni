Overview

We add our own type of WEBrick::HTTPProxyServer class that supports notifications when the user tries to access a resource irrelevant to the scan and does not restrict header exchange.

def choose_header(src, dst)
    connections = split_field(src['connection'])
    src.each{|key, value|
        key = key.downcase
        if HopByHop.member?(key)          || # RFC2616: 13.5.1
           connections.member?(key)       || # RFC2616: 14.10
           # ShouldNotTransfer.member?(key)    # pragmatics
          @logger.debug("choose_header: `#{key}: #{value}'")
          next
        end
    dst[key] = value
  }
end

def notify( reasons, req, res )
    res.header['content-type'] = 'text/plain'
    res.header.delete( 'content-encoding' )

    res.body << reasons.map{ |msg| " *  #{msg}" }.join( "\n" )
end

def service( req, res )
    exclude_reasons = @config[:ProxyURITest].call( req.unparsed_uri )

    if( exclude_reasons.empty? )
        super( req, res )
    else
        notify( exclude_reasons, req, res )
    end
end
