def run
    # will hold the hash IDs of inconclusive issues
    inconclusive = []
    @framework.audit_store.issues.each.with_index {
        |issue, idx|
        next if !issue.verification

        inconclusive << {
            'hash'   => issue._hash,
            'index'  => idx + 1,
            'url'    => issue.url,
            'name'   => issue.name,
            'var'    => issue.var,
            'elem'   => issue.elem,
            'method' => issue.method
        }
    }

    register_results( inconclusive ) if !inconclusive.empty?
end
