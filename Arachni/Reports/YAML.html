def run
    print_line( )
    print_status( 'Dumping audit results in \'' + @options['outfile']  + '\'.' )

    File.open( @options['outfile'], 'w' ) {
        |f|
        f.write( ::YAML::dump( @audit_store.to_h ) )
    }

    print_status( 'Done!' )
end
