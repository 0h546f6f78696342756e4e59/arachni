def __add( text, no_escape = false )

        __buffer( CGI.escapeHTML( text ) )
    else
        __buffer( text )
    end

def add_headers( type, headers )

    start_tag( type )
    headers.each_pair {
        |name, value|
        __buffer( "<field name=\"#{name}\" value=\"#{CGI.escapeHTML( value.strip )}\" />" )
    }
    end_tag( type )
end
