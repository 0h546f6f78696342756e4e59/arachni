def run( )

    print_line( )
    print_status( 'Creating HTML report...' )

    @template = Liquid::Template.parse( IO.read( @tpl ) )
    
    out = @template.render( __prepare_data( ) )
    
    __save( @outfile, out )

    print_status( 'Saved in \'' + @outfile + '\'.' )
end
