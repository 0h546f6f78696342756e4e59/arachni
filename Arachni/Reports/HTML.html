def run( )

    print_line( )
    print_status( 'Creating HTML report...' )

    @template = Liquid::Template.parse( IO.read( @options['tpl'] ) )

    out = @template.render( __prepare_data( ) )

    __save( @options['outfile'], out )

    print_status( 'Saved in \'' + @options['outfile'] + '\'.' )
end
