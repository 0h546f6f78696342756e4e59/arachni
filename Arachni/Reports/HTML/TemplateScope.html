def find_issue_by_hash( hash )
    @audit_store.issues.each_with_index {
        |issue, i|
        return [i+1, issue] if issue._hash == hash
    }
    return nil
end

def normalize( str )
    return '' if !str || str.empty?
    str.encode( 'UTF-8', :invalid => :replace, :undef => :replace )
end
