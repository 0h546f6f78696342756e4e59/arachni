def prepare

    @@__injection_str ||= []

    if @@__injection_str.empty?
        read_file( 'payloads.txt' ) {
            |str|

            [ ' ', ' && ', ';' ].each {
                |sep|
                @@__injection_str << sep + " " + str
            }

        }
    end

    @__opts = {
        :format  => [ Format::STRAIGHT ],
        :timeout => 4000
    }
end
