def normalize(value)
    if (value.to_s.match(/^0x[a-fA-F\d]+$/))
        value.to_i(16)
    else
        value.to_i
    end

def valid?(value)
    return false if empty_required_value?(value)

    if value and not normalize(value).to_s.match(/^\d+$/)
        return false
    end

    return super
end
