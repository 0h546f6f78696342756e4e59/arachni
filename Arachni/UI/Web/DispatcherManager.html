def connect( url )
    @@cache ||= {}

    begin
        if @@cache[url] && @@cache[url].alive?
            return @@cache[url]
        elsif ( tmp = Arachni::RPC::XML::Client::Dispatcher.new( @opts, url ) ) &&
              tmp.alive?
            return @@cache[url] = tmp
        end
    rescue Exception => e
      return nil
    end
end
