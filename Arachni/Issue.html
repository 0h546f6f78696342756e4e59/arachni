def initialize( opts = {} )
    @verification = false

    opts.each {
        |k, v|
        begin
            send( "#{k.to_s.downcase}=", encode( v ) )
        rescue Exception => e
        end
    }

    opts[:issue].each {
        |k, v|
        begin
            send( "#{k.to_s.downcase}=", encode( v ) )
        rescue Exception => e
        end
    } if opts[:issue]

    if( @cwe )
        @cwe_url = "http://cwe.mitre.org/data/definitions/" + @cwe + ".html"
    end

    @mod_name   = opts[:name]
    @references = opts[:references] || {}
end
