def lsplug

    plug_info = []

    @plugins.available( ).each {
        |plugin|

        info = @plugins[plugin].info

        info[:plug_name]   = plugin
        info[:path]        = @plugins.name_to_path( plugin )

        info[:options] = [info[:options]].flatten.compact.map {
            |opt|
            opt_h = opt.to_h
            opt_h['default'] = '' if opt_h['default'].nil?
            opt_h['type']    = opt.type
            opt_h
        }

        plug_info << info
    }

    @plugins.clear( )

    return plug_info
end
