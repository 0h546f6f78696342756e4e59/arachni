def initialize(file)
  raise "TokyoCabinet filename must have .tch extension" if File.extname(file) != '.tch'
  @db = ::TokyoCabinet::HDB::new
  @db.open(file, ::TokyoCabinet::HDB::OWRITER | ::TokyoCabinet::HDB::OCREAT)
  @db.clear
end
