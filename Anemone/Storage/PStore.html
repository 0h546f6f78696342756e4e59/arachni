def each
  @keys.each_key do |key|
    value = nil
    @store.transaction { |s| value = s[key] }
    yield key, value
  end
end

def merge!(hash)
  @store.transaction do |s|
    hash.each { |key, value| s[key] = value; @keys[key] = nil }
  end
  self
end
