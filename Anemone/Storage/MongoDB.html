def []=(url, page)
  hash = page.to_hash
  BINARY_FIELDS.each do |field|
    hash[field] = BSON::Binary.new(hash[field]) unless hash[field].nil?
  end
  @collection.update(
    {'url' => page.url.to_s},
    hash,
    :upsert => true
  )
end

def each
  @collection.find do |cursor|
    cursor.each do |doc|
      page = load_page(doc)
      yield page.url.to_s, page 
    end
  end
end
